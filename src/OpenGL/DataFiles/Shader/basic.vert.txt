#version 410 core
layout (location = 0) in vec3 Position;
layout (location = 1) in vec3 Normal;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
out vec3 vPosition;
out vec3 vNormal;

void main(){

vec4 p = model * vec4(Position, 1.0f);
vPosition = (p / p.w).xyz;
vec4 n = model * vec4(Position, 1.0f);
vNormal = (n / n.w).xyz;

}
