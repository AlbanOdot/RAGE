Build instructions :

cd RAGE/Tools/OpenMesh-7.1/build

cmake .. -DCMAKE_INSTALL_PREFIX="THE ABSOLUTE PATH TO THE install SUBDIRECTORY"

make -j

make install

cd ../../..

mkdir Releas

cd Release

cmake ..
